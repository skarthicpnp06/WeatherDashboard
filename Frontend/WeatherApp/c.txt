import React, { useState } from 'react'
import { registerWeatherAlert } from '../Services/weatherservice'
import Errormessage from '../Components/Errormessage'
import '../Styles/dashboard.css'

const AlertSetup = () => {
  const [email, setEmail] = useState("")
  const [city, setCity] = useState("")
  const [targetTemp, setTargetTemp] = useState("")
  const [condition, setCondition] = useState("ABOVE")
  const [message, setMessage] = useState("")
  const [error, setError] = useState("")
  const [loading, setLoading] = useState(false)

  const handleSubmit = async (e) => {
    e.preventDefault()
    setError("")
    setMessage("")

    if (!email || !city || !targetTemp) {
      setError("Please fill out all registration fields.")
      return
    }

    try {
      setLoading(true)
      const payload = {
        email: email.trim(),
        city: city.trim(),
        targetTemp: parseFloat(targetTemp),
        triggerCondition: condition
      }
      
      await registerWeatherAlert(payload)
      setMessage(`🎉 Notification successfully registered! We'll track ${city} hourly.`)
      setEmail("")
      setCity("")
      setTargetTemp("")
    } catch (err) {
      setError("Failed to create alert profile. Check connection.")
    } finally {
      setLoading(false)
    }
  }

  return (
    <div className="container" style={{ maxWidth: '450px' }}>
      <h2 className="title">⏰ Weather Triggers</h2>
      <p style={{ fontSize: '13px', color: '#666', textAlign: 'center', marginBottom: '20px' }}>
        Get an automatic email notification when a city passes your set threshold temperature.
      </p>

      <form onSubmit={handleSubmit} className="alert-form">
        <input 
          type="email" 
          placeholder="Enter your Email Address" 
          value={email}
          onChange={(e) => setEmail(e.target.value)}
        />
        
        <input 
          type="text" 
          placeholder="Target City Name" 
          value={city}
          onChange={(e) => setCity(e.target.value)}
        />

        <div className="alert-rule-row">
          <select value={condition} onChange={(e) => setCondition(e.target.value)}>
            <option value="ABOVE">If Temp is ABOVE (☀️)</option>
            <option value="BELOW">If Temp is BELOW (❄️)</option>
          </select>
          
          <input 
            type="number" 
            placeholder="°C" 
            value={targetTemp}
            onChange={(e) => setTargetTemp(e.target.value)}
            style={{ width: '80px' }}
          />
        </div>

        <button type="submit" disabled={loading}>
          {loading ? "Subscribing..." : "Activate Alarm Trigger"}
        </button>
      </form>

      {error && <Errormessage message={error} />}
      {message && <div className="success-box">{message}</div>}
    </div>
  )
}

export default AlertSetup

import React, { useState, useEffect } from 'react'

import { LineChart, Line, BarChart, Bar, XAxis, YAxis, CartesianGrid, Tooltip, Legend, ResponsiveContainer } from 'recharts'
import '../Styles/dashboard.css'

const Analytics = ({ currentCity }) => {
  const [historyData, setHistoryData] = useState([])
  const [error, setError] = useState("")
  const [loading, setLoading] = useState(false)

  useEffect(() => {
    if (!currentCity) {
      setError("Please search for a city on the dashboard first to load its search trend analytics.")
      return
    }

    const fetchHistory = async () => {
      try {
        setLoading(true)
        setError("")
        const data = await getWeatherHistory(currentCity)
        
        // Format the database records for cleaner chart labels
        const formattedData = data.map((item, index) => ({
          name: `Search #${index + 1}`,
          Temp: Math.round(item.temparature),
          Humidity: item.humidity,
          WindSpeed: item.windspeed
        }))
        
        setHistoryData(formattedData)
      } catch (err) {
        setError(err.message)
      } finally {
        setLoading(false)
      }
    }

    fetchHistory()
  }, [currentCity])

  return (
    <div className="container" style={{ maxWidth: '800px' }}>
      <h2 className="title">📊 Cache Insights: <span style={{ color: '#007bff', textTransform: 'capitalize' }}>{currentCity || "None"}</span></h2>
      
      {loading && <div className="loader">Analyzing database logs...</div>}
      {error && <Errormessage message={error} />}

      {!loading && !error && historyData.length === 0 && (
        <p style={{ textAlign: 'center', color: '#666' }}>No historical metrics stored for this city yet.</p>
      )}

      {!loading && !error && historyData.length > 0 && (
        <div className="analytics-layout">
          <p className="analytics-subtitle">Tracking data shifts captured across past searches in your MySQL database cache:</p>
          
          {/* CHART 1: Temperature Trajectory Line Graph */}
          <div className="chart-wrapper">
            <h3>🌡️ Temperature Trajectory (°C)</h3>
            <ResponsiveContainer width="100%" height={250}>
              <LineChart data={historyData} margin={{ top: 10, right: 30, left: 0, bottom: 0 }}>
                <CartesianGrid strokeDasharray="3 3" />
                <XAxis dataKey="name" stroke="#666" />
                <YAxis stroke="#666" />
                <Tooltip />
                <Legend />
                <Line type="monotone" dataKey="Temp" stroke="#ff4500" strokeWidth={3} activeDot={{ r: 8 }} />
              </LineChart>
            </ResponsiveContainer>
          </div>

          {/* CHART 2: Humidity & Wind Metrics Bar Graph */}
          <div className="chart-wrapper" style={{ marginTop: '30px' }}>
            <h3>💧 Humidity vs 💨 Wind Speed Trends</h3>
            <ResponsiveContainer width="100%" height={250}>
              <BarChart data={historyData} margin={{ top: 10, right: 30, left: 0, bottom: 0 }}>
                <CartesianGrid strokeDasharray="3 3" />
                <XAxis dataKey="name" stroke="#666" />
                <YAxis stroke="#666" />
                <Tooltip />
                <Legend />
                <Bar dataKey="Humidity" fill="#007bff" radius={[4, 4, 0, 0]} />
                <Bar dataKey="WindSpeed" fill="#28a745" radius={[4, 4, 0, 0]} />
              </BarChart>
            </ResponsiveContainer>
          </div>
        </div>
      )}
    </div>
  )
}

import React, { useState } from 'react'

const Compare = () => {
  const [city1, setCity1] = useState("")
  const [city2, setCity2] = useState("")
  const [weather1, setWeather1] = useState(null)
  const [weather2, setWeather2] = useState(null)
  const [error, setError] = useState("")
  const [loading, setLoading] = useState(false)

  const handleCompare = async (e) => {
    e.preventDefault()
    if (!city1.trim() || !city2.trim()) {
      alert("Please enter both city names to compare.")
      return
    }

    try {
      setError("")
      setLoading(true)
      setWeather1(null)
      setWeather2(null)

      // Fetch both APIs simultaneously to minimize loading screen lag
      const [data1, data2] = await Promise.all([
        getWeatherData(city1.trim()),
        getWeatherData(city2.trim())
      ])

      setWeather1(data1)
      setWeather2(data2)
    } catch (err) {
      setError("Failed to fetch data for one or both cities. Check your spelling.")
    } finally {
      setLoading(false)
    }
  }

  //  Visual Helper to highlight the warmer city
  const isCity1Warmer = weather1 && weather2 && weather1.temparature > weather2.temparature
  const isCity2Warmer = weather1 && weather2 && weather2.temparature > weather1.temparature

  return (
    <div className="container compare-container">
      <h2 className="title">⚔️ City Comparision</h2>
      
      <form onSubmit={handleCompare} className="compare-search-box">
        <input 
          type="text" 
          placeholder="First City" 
          value={city1} 
          onChange={(e) => setCity1(e.target.value)}
        />
        <span className="vs-badge">VS</span>
        <input 
          type="text" 
          placeholder="Second City" 
          value={city2} 
          onChange={(e) => setCity2(e.target.value)}
        />
        <button type="submit">Compare</button>
      </form>

      {loading && <div className="loader">Analyzing climates...</div>}
      {error && <Errormessage message={error} />}

      {!loading && weather1 && weather2 && (
        <div className="compare-grid">
          <div className={`compare-column ${isCity1Warmer ? 'warmer-highlight' : ''}`}>
            {isCity1Warmer && <div className="winner-tag">🔥 Warmer</div>}
            <WeatherCard weather={weather1} />
          </div>
          
          <div className={`compare-column ${isCity2Warmer ? 'warmer-highlight' : ''}`}>
            {isCity2Warmer && <div className="winner-tag">🔥 Warmer</div>}
            <WeatherCard weather={weather2} />
          </div>
        </div>
      )}
    </div>
  )
}

export default Compare

const Forecast = ({ currentCity }) => {
  const [forecastData, setForecastData] = useState(null)
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState(null)

  useEffect(() => {
    if (!currentCity) return

    const fetchForecast = async () => {
      setLoading(true)
      setError(null)
      try {
        const data = await getWeatherForecast(currentCity)
        setForecastData(data)
      } catch (err) {
        setError(err.message)
      } finally {
        setLoading(false)
      }
    }

    fetchForecast()
  }, [currentCity])

  if (!currentCity) {
    return (
      <div style={{ textAlign: 'center', padding: '40px', color: 'white' }}>
        <h3>🔍 Please search for a city on the Current Weather tab first!</h3>
      </div>
    )
  }

  if (loading) return <p style={{ color: 'white', textAlign: 'center', padding: '20px' }}>Loading 5-day forecast data...</p>
  if (error) return <p style={{ color: '#ff6b6b', textAlign: 'center', padding: '20px' }}>Error: {error}</p>

  const forecastList = forecastData?.list || []

 
  const formatTimeStr = (str) => {
    const date = new Date(str)
    return date.toLocaleDateString('en-US', { 
      weekday: 'short', 
      month: 'short', 
      day: 'numeric',
      hour: '2-digit',
      minute: '2-digit'
    })
  }

  return (
    <div style={{ padding: '10px 20px' }}>
      <h2 style={{ color: 'white', marginBottom: '20px', textAlign: 'center' }}>
        📅 5-Day Forecast: {currentCity.toUpperCase()}
      </h2>
      
      
      <div style={styles.forecastGrid}>
        {forecastList.map((item, index) => (
          <div className="weather-card" key={index} style={styles.cardOverride}>
            <p style={{ fontSize: '13px', fontWeight: 'bold', color: '#555', margin: '0 0 8px 0' }}>
              ⏰ {formatTimeStr(item.dt_txt)}
            </p>
            
            <h3 style={{ fontSize: '28px', margin: '5px 0', color: '#222' }}>
              {Math.round(item.main.temp)}°C
            </h3>
            
            <p style={{ fontStyle: 'italic', color: '#666', fontSize: '14px', margin: '0 0 10px 0' }}>
              ✨ {item.weather[0].description}
            </p>
            
            <hr style={{ border: 'none', borderTop: '1px solid #eee', margin: '10px 0' }} />
            
            <div style={styles.smallMetrics}>
              <div>🤔 Feels: {Math.round(item.main.feels_like)}°C</div>
              <div>💧 Hum: {item.main.humidity}%</div>
            </div>
          </div>
        ))}
      </div>
    </div>
  )
}

// The base API URL for your Spring Boot local backend server
const BASE_URL = "http://localhost:8080/weather"; 

export async function getWeatherForecast(city) {
  // 🚨 Make sure this endpoint appends /forecast!
  const response = await fetch(`${BASE_URL}/forecast?city=${city}`);
  if (!response.ok) {
    throw new Error("Failed to load 5-day forecast array properties from backend server");
  }
  return await response.json();
}
 
export async function getWeatherData(city) {
  const response = await fetch(`${BASE_URL}?city=${city}`);
  if (!response.ok) {
    throw new Error("Failed to load current weather data");
  }
  return await response.json();
}


export async function getWeatherHistory(city) {
  const response = await fetch(`${BASE_URL}/history?city=${city}`);
  if (!response.ok) {
    throw new Error("Failed to load historical cache data");
  }
  return await response.json();
}


export async function registerWeatherAlert(alertPayload) {
  const response = await fetch(`${BASE_URL}/alerts`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(alertPayload)
  });
  if (!response.ok) {
    throw new Error("Failed to subscribe to notifications");
  }
  return await response.json();
}

package com.example.Backend.Service;

import java.util.List;
import java.util.Properties;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.scheduling.annotation.Scheduled;
import org.springframework.stereotype.Service;

import com.example.Backend.Model.AlertEntity;
import com.example.Backend.Model.WeatherEntity;
import com.example.Backend.Repository.AlertRepository;

import jakarta.mail.Message;

import jakarta.mail.Transport;
import jakarta.mail.internet.InternetAddress;
import jakarta.mail.internet.MimeMessage;

@Service
public class AlertService {

    @Autowired
    private AlertRepository alertRepository;

    @Autowired
    private WeatherService weatherService;

    @Value("${spring.mail.username}")
    private String username;

    @Value("${spring.mail.password}")
    private String password;

    @Value("${spring.mail.host}")
    private String host;

    @Value("${spring.mail.port}")
    private String port;

    public AlertEntity saveAlert(AlertEntity alert) {
    String cleanCity = alert.getCity().trim().toLowerCase();
    String cleanEmail = alert.getEmail().trim().toLowerCase();
    
    // Look through all alerts to see if this user already tracking this city
    List<AlertEntity> allAlerts = alertRepository.findAll();
    for (AlertEntity existing : allAlerts) {
        if (existing.getEmail().equalsIgnoreCase(cleanEmail) && existing.getCity().equalsIgnoreCase(cleanCity)) {
            
            existing.setTargetTemp(alert.getTargetTemp());
            existing.setTriggerCondition(alert.getTriggerCondition());
            return alertRepository.save(existing);
        }
    }
    
    
    alert.setCity(cleanCity);
    alert.setEmail(cleanEmail);
    return alertRepository.save(alert);
}


    @Scheduled(fixedRate = 3600000) 
public void checkWeatherAlerts() {
    List<AlertEntity> activeAlerts = alertRepository.findAll();
    
    // Track processed email+city combinations to prevent duplicate sends in the same hour
    java.util.HashSet<String> processedAlerts = new java.util.HashSet<>();
    
    for (AlertEntity alert : activeAlerts) {
        String uniqueKey = alert.getEmail().toLowerCase() + "-" + alert.getCity().toLowerCase();
        
        // If we already sent an email to this user for this city in this loop, skip it!
        if (processedAlerts.contains(uniqueKey)) {
            continue; 
        }

        try {
            WeatherEntity currentLiveWeather = weatherService.getWeather(alert.getCity());
            double currentTemp = currentLiveWeather.getTemparature();
            boolean triggered = false;

            if (alert.getTriggerCondition().equalsIgnoreCase("ABOVE") && currentTemp > alert.getTargetTemp()) {
                triggered = true;
            } else if (alert.getTriggerCondition().equalsIgnoreCase("BELOW") && currentTemp < alert.getTargetTemp()) {
                triggered = true;
            }

            if (triggered) {
                sendNotificationEmail(alert.getEmail(), alert.getCity(), currentTemp, alert.getTargetTemp(), alert.getTriggerCondition());
                
                processedAlerts.add(uniqueKey); 
            }
        } catch (Exception e) {
            System.out.println("Could not process alert check for ID " + alert.getId() + ": " + e.getMessage());
        }
    }
}

    private void sendNotificationEmail(String toEmail, String city, double currentTemp, double targetTemp, String condition) {
        // Setup SMTP server properties using your application.properties settings
        Properties props = new Properties();
        props.put("mail.smtp.auth", "true");
        props.put("mail.smtp.starttls.enable", "true");
        props.put("mail.smtp.host", host);
        props.put("mail.smtp.port", port);

        // Session authentication authentication
        Session session = Session.getInstance(props, new jakarta.mail.Authenticator() {
            protected PasswordAuthentication getPasswordAuthentication() {
                return new PasswordAuthentication(username, password);
            }
        });

        try {
            Message message = new MimeMessage(session);
            message.setFrom(new InternetAddress(username));
            message.setRecipients(Message.RecipientType.TO, InternetAddress.parse(toEmail));
            message.setSubject("🚨 SkySync Weather Alert: " + city.toUpperCase());
            
            String emailText = String.format(
                "Hello!\n\nYour SkySync weather trigger has fired.\n" +
                "The current temperature in %s is %s°C, which is %s your threshold target of %s°C.\n\n" +
                "Stay prepared!\n- SkySync Weather Engine",
                city.toUpperCase(), currentTemp, condition.toLowerCase(), targetTemp
            );
            
            message.setText(emailText);
            Transport.send(message);
            System.out.println("📬 Alert email successfully dispatched to " + toEmail);

        } catch (MessagingException e) {
            throw new RuntimeException("Failed to transmit email alert", e);
        }
    }
}

import java.util.List;
import java.util.Map;
import java.util.function.Function;
import org.springframework.web.client.HttpClientErrorException;
import org.springframework.http.HttpStatus;

@Service
public class WeatherService {

    @Autowired
    private WeatherRepositoryextends weatherRepository;

    private final RestTemplate restTemplate = new RestTemplate();

    // API Keys
    private final String OPENWEATHER_KEY = "f5e923868577018c7a11ff3623dafb41";
    private final String WEATHERAPI_KEY = "4d2f4f3e0060493d95860505262705";

    public WeatherEntity getWeather(String city) {

    if(city==null)  return null ;
    String cleancity=city.trim().toLowerCase();
    // 1. Try Primary API (OpenWeather)
    try {
        String url = "https://api.openweathermap.org/data/2.5/weather?q=" + cleancity + "&appid=" + OPENWEATHER_KEY + "&units=metric";
        return fetchAndSave(url, response -> mapOpenWeather(response));
    } catch (HttpClientErrorException e) {
        // If the API explicitly says the city wasn't found, stop immediately and pass it to React
        if (e.getStatusCode() == HttpStatus.NOT_FOUND) {
            throw e; 
        }
        System.out.println("OpenWeather HTTP error: " + e.getMessage());
    } catch (Exception e) {
        System.out.println("OpenWeather failed: " + e.getMessage());
    }

    // 2. Try Secondary API (WeatherAPI)
    try {
        String url = "https://api.weatherapi.com/v1/current.json?key=" + WEATHERAPI_KEY + "&q=" + cleancity;
        return fetchAndSave(url, response -> mapWeatherApi(response));
    } catch (HttpClientErrorException e) {
        if (e.getStatusCode() == HttpStatus.NOT_FOUND) {
            throw e;
        }
        System.out.println("WeatherAPI HTTP error: " + e.getMessage());
    } catch (Exception e) {
        System.out.println("WeatherAPI failed: " + e.getMessage());
    }

    // 3. Try Database Cache Fallback
    try {
        WeatherEntity cachedData = weatherRepository.findTopByCityOrderByIdDesc(cleancity);
        if (cachedData != null) {
            cachedData.setSource("Database Fallback");
            return cachedData;
        }
    } catch (Exception e) {
        System.out.println("Cache read failed: " + e.getMessage());
    }

    // 4. Ultimate Static Fallback (Only reached if network is down but city might be valid)
    return new WeatherEntity(city, 30.0, "Fallback Data", 70, 10.0, "Static Fallback");
}
    /**
 * Generic helper method to fetch data from an API and save it as a 
 * brand-new historical record in the database every single time.
 */
private WeatherEntity fetchAndSave(String url, Function<Map, WeatherEntity> mapper) {
    // 1. Fetch live data from third-party weather API
    Map response = restTemplate.getForObject(url, Map.class);
    if (response == null) throw new RuntimeException("Empty API response");
    
    // 2. Map the raw API JSON response into a fresh WeatherEntity object
    WeatherEntity entity = mapper.apply(response);
    
    // 3. Save it directly. Without an attached ID, Hibernate will ALWAYS insert a fresh row!
    return weatherRepository.save(entity);
}
    private WeatherEntity mapOpenWeather(Map response) {
        Map main = (Map) response.get("main");
        Map wind = (Map) response.get("wind");
        List weatherList = (List) response.get("weather");
        Map weatherData = (Map) weatherList.get(0);

        return new WeatherEntity(
                (String) response.get("name"),
                ((Number) main.get("temp")).doubleValue(),
                (String) weatherData.get("main"),
                ((Number) main.get("humidity")).intValue(),
                ((Number) wind.get("speed")).doubleValue(),
                "OpenWeather API"
        );
    }

    /**
     * Parsing logic specifically for WeatherAPI structure
     */
    private WeatherEntity mapWeatherApi(Map response) {
        Map location = (Map) response.get("location");
        Map current = (Map) response.get("current");
        Map conditionMap = (Map) current.get("condition");

        return new WeatherEntity(
                (String) location.get("name"),
                ((Number) current.get("temp_c")).doubleValue(),
                (String) conditionMap.get("text"),
                ((Number) current.get("humidity")).intValue(),
                ((Number) current.get("wind_kph")).doubleValue(),
                "WeatherAPI (Secondary)"
        );
    }

    public List<WeatherEntity> getWeatherforecast(String city) {
        List<WeatherEntity> forecastList = new ArrayList<>();
        try {
        String url = "https://api.openweathermap.org/data/2.5/forecast?q=" + city + "&appid=" + OPENWEATHER_KEY + "&units=metric";
        Map response = restTemplate.getForObject(url, Map.class);
        
        if (response != null && response.containsKey("list")) {
            List list = (List) response.get("list");
            String cityName = (String) ((Map) response.get("city")).get("name");

            // OpenWeather gives 3-hour intervals. Skipping by 8 grabs 1 forecast per day (3h * 8 = 24h)
            for (int i = 0; i < list.size(); i += 8) {
                Map dayData = (Map) list.get(i);
                Map main = (Map) dayData.get("main");
                Map wind = (Map) dayData.get("wind");
                List weatherArray = (List) dayData.get("weather");
                Map weatherData = (Map) weatherArray.get(0);
                
                // We use the text date string parsed from the API to display the day info
                String dtTxt = (String) dayData.get("dt_txt"); 
                String shortDate = dtTxt.split(" ")[0]; // Splitting to get YYYY-MM-DD

                WeatherEntity forecastEntry = new WeatherEntity(
                        cityName,
                        ((Number) main.get("temp")).doubleValue(),
                        (String) weatherData.get("main"),
                        ((Number) main.get("humidity")).intValue(),
                        ((Number) wind.get("speed")).doubleValue(),
                        shortDate // Storing the target date string temporarily in the source field for UI display
                );
                forecastList.add(forecastEntry);
            }
        }
    } catch (Exception e) {
        System.out.println("Forecast fetch failed: " + e.getMessage());
        // Simple fallback dummy loop if forecast API down
        for (int i = 1; i <= 5; i++) {
            forecastList.add(new WeatherEntity(city, 25.0 + i, "Clear", 60, 5.0, "Day +" + i));
        }
    }
    return forecastList;
}
